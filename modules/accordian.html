function accordion_toggel(list_no,index){
    var acc_title = document.getElementsByName('acc'+list_no+'t');
    var acc_body = document.getElementsByName('acc'+list_no+'b');
    var acc_arrow = document.getElementsByName('acc'+list_no+'ar');
    if(acc_title[index-1].className != "accordion_title active"){
        acc_title[index-1].className = "accordion_title active";
        acc_body[index-1].style.marginTop = "0px";
        acc_arrow[index-1].style.transform = "rotate(180deg)";
        acc_arrow[index-1].style.filter = "invert(100%)";
    }
    else{
        acc_title[index-1].className = "accordion_title";
        acc_body[index-1].style.marginTop = "-300px";
        acc_arrow[index-1].style.transform = "rotate(0deg)";
        acc_arrow[index-1].style.filter = "invert(49%) sepia(51%) saturate(744%) hue-rotate(326deg) brightness(90%) contrast(108%)";
    }
}
